function ##f(n)
    n = sconvert2int(n)
    _sprintout(splus(splus(NiString("**** f("), scast2string(sref(n))), NiString(") called")))
    if sasbool(sgreater(sref(n), 0))
        return (Unknown("g"))(scopy(sminus(sref(n), 1))...)
    end
    return nothing
end
